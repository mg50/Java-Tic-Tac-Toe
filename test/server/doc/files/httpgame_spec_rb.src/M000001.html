        def start_two_player_game
                game1 = HTTPGame.new
                game2 = HTTPGame.new

                [game1, game2].each do |game|
                        game.start nil
                        game.receive_signal "signal" => "YES" # for 3x3
                        game.receive_signal "signal" => "NO"  # for 2-strategy game
                end

                [game1, game2]
        end
